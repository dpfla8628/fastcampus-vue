- html
```html
<div id="app">
  <ul>  
<my-todo-item v-for="todo in todos"
              b-bind:key="todo.id"
              v-bind:todo="todo"></my-todo-item>
  </ul>
</div>
```
- css
```css
li span.done{
  text-decoration : line-through;
}
```
- js
```js
Vue.component('my-todo-item',{
  props : ['todo'],
  template:`
      <li>
      <input type="checkbox" v-model="todo.done">
      <span v-bind:class="{done : todo.done}">            {{todo.title}}
      </span>
      </li>
  `
})

const vm = new Vue({
  el : '#app',
  data : {
    todos : [
      {
        id:'1',
        title:'아침 먹기',
        done:true
      },
     {
        id:'2',
        title:'점심 먹기',
        done:false
      },
      {
        id:'3',
        title:'저녁 먹기',
        done:false
      },
      {
        id:'4',
        title:'야식 먹기',
        done:false
      },
    ]
  }
})
```
